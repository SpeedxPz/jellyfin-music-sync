── src/main/lib/fs-utils.ts ───────────────────────────────────────────────────
sanitizePathSegment wraps the sanitize-filename package with an empty-string fallback.
atomicWriteJson MUST write .tmp to the same directory as the target (not os.tmpdir()) to avoid EXDEV cross-device errors.

```typescript
// src/main/lib/fs-utils.ts
import sanitize from 'sanitize-filename'
import { writeFileSync, renameSync, readFileSync, mkdirSync } from 'fs'
import { dirname } from 'path'

/**
 * Sanitize a single path segment for FAT32 compatibility.
 *
 * APPLY PER SEGMENT — never to full paths containing slashes.
 * Correct: path.join(sanitizePathSegment(artist), sanitizePathSegment(album), sanitizePathSegment(file))
 * Wrong:   sanitizePathSegment(`${artist}/${album}/${file}`)
 *
 * Handles:
 * - Illegal chars: \ / : * ? " < > | and control chars 0x00–0x1F
 * - Windows reserved names: CON, PRN, AUX, NUL, COM0–9, LPT0–9 (with or without extension)
 * - Trailing dots and trailing spaces
 * - 255-byte length limit
 *
 * Returns '_' for empty or whitespace-only input.
 */
export function sanitizePathSegment(segment: string): string {
  const sanitized = sanitize(segment, { replacement: '_' })
  if (!sanitized || sanitized.trim() === '') {
    return '_'
  }
  return sanitized
}

/**
 * Write JSON atomically: write to <path>.tmp, then rename to final path.
 *
 * The .tmp file is always in the same directory as the target — never in
 * os.tmpdir() — to guarantee same-volume placement and avoid EXDEV errors.
 */
export function atomicWriteJson(filePath: string, data: unknown): void {
  const tmpPath = `${filePath}.tmp`
  const dir = dirname(filePath)
  mkdirSync(dir, { recursive: true })
  writeFileSync(tmpPath, JSON.stringify(data, null, 2), 'utf-8')
  // renameSync is atomic on same-volume writes.
  // .tmp is co-located with target — EXDEV cannot occur.
  renameSync(tmpPath, filePath)
}

/**
 * Read and parse a JSON file, returning fallback on any error.
 * Handles missing files, corrupt JSON, and permission errors.
 */
export function safeReadJson<T>(filePath: string, fallback: T): T {
  try {
    const raw = readFileSync(filePath, 'utf-8')
    return JSON.parse(raw) as T
  } catch {
    return fallback
  }
}
```

```typescript
// src/main/ipc/settings.ts
import { ipcMain } from 'electron'
import { store } from '../lib/store'
import { getLogPath } from '../lib/logger'
import type { Settings } from '../../../shared/ipc-types'

export function registerSettingsHandlers(): void {
  // SET-01 + SET-02: Get current settings (lastDestination + concurrentDownloads)
  ipcMain.handle('settings:get', (): Settings => {
    return store.store
  })

  // SET-01 + SET-02: Update settings (partial merge; concurrentDownloads clamped 1–5)
  ipcMain.handle('settings:set', (_evt, partial: Partial<Settings>): void => {
    if (partial.concurrentDownloads !== undefined) {
      partial = {
        ...partial,
        concurrentDownloads: Math.max(1, Math.min(5, partial.concurrentDownloads)),
      }
    }
    store.set(partial)
  })

  // SET-03: Expose log file path to renderer (app.getPath is main-process-only)
  ipcMain.handle('settings:getLogPath', (): string => {
    return getLogPath()
  })
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| IPC handler input (renderer → main) | Partial<Settings> values arrive from renderer context; must be validated/clamped before reaching store |
| Filesystem (settings store) | electron-conf stores JSON in userData; a tampered JSON file could inject malformed values |
| Filesystem (log file) | log messages are written from main process only; renderer cannot write directly |
| Path segments (Jellyfin data → filesystem) | Server-provided strings must be sanitized before use as path components |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | settings:set IPC handler | mitigate | concurrentDownloads is clamped to Math.max(1, Math.min(5, value)) in registerSettingsHandlers() before store.set(). Schema validation is a second layer. Renderer cannot send arbitrary values that escape the 1–5 range. |
| T-03-02 | Tampering | electron-conf userData JSON | mitigate | electron-conf validates against schema on read; corrupt or injected values fail validation and fall back to defaults. Files are in app.getPath('userData') which requires OS-level access. |
| T-03-03 | Tampering | Path traversal via sanitizePathSegment | mitigate | sanitizePathSegment() wraps sanitize-filename which strips / \ : * ? " < > | and control chars. Applied per segment (never to full paths). Acceptance criteria verify AC/DC → AC_DC and reserved-name stripping. |
| T-03-04 | Information Disclosure | Debug log file | accept | Log contains operational messages (INFO/WARN/ERROR). Plan 03 logs only 'App started'. Future phases must not log credentials (auth tokens, passwords) — this constraint is noted for Phase 2 planning. |
| T-03-05 | Denial of Service | Corrupt userData JSON | mitigate | safeReadJson() wraps parse in try/catch and returns fallback. atomicWriteJson() prevents corrupt writes via .tmp + renameSync. Store survives crash without corruption. |
| T-03-06 | Elevation of Privilege | Stub channels returning null | mitigate | D-06: All stubs throw new Error('Not implemented: <channel>'). Silent return null/undefined is forbidden — verified by grep for 'Not implemented:' in stubs.ts acceptance criteria. |
</threat_model>

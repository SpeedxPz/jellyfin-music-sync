  <behavior>
sanitizePathSegment:
- 'AC/DC' → 'AC_DC' (forward slash stripped, replaced with underscore)
- 'CON' → not 'CON' (Windows reserved name, any non-CON non-empty string is acceptable)
- 'NUL.flac' → not 'NUL.flac' (reserved name with extension is still reserved)
- 'CON.mp3' → not 'CON.mp3' (same rule)
- '...And Justice For All' → result starts with '...' (leading dots preserved)
- 'track   ' → 'track' (trailing spaces stripped)
- '' → '_' (empty string returns fallback)
- '   ' → '_' (whitespace-only returns fallback)

atomicWriteJson + safeReadJson:
- round-trips: atomicWriteJson(path, data); safeReadJson(path, null) deep-equals data
- creates intermediate directories: atomicWriteJson('a/b/c/d.json', ...) succeeds
- missing file: safeReadJson('/nonexistent/path.json', 'fallback') → 'fallback'
- corrupt JSON: after writing corrupt bytes, safeReadJson returns fallback (not throw)
  </behavior>

Tests import directly from ../../src/main/lib/fs-utils. No Electron mock required — fs-utils only imports sanitize-filename and Node.js builtins (fs, path). Tests use os.tmpdir() for temporary directories via mkdtempSync; each test suite cleans up in afterEach.

```typescript
// tests/lib/fs-utils.test.ts
import { describe, it, expect, beforeEach, afterEach } from 'vitest'
import { mkdtempSync, rmSync, writeFileSync } from 'fs'
import { join } from 'path'
import { tmpdir } from 'os'
import { sanitizePathSegment, atomicWriteJson, safeReadJson } from '../../src/main/lib/fs-utils'

describe('sanitizePathSegment', () => {
  it('strips forward slash: AC/DC → AC_DC', () => {
    expect(sanitizePathSegment('AC/DC')).toBe('AC_DC')
  })

  it('sanitizes Windows reserved name: CON → not CON', () => {
    const result = sanitizePathSegment('CON')
    expect(result).not.toBe('CON')
    expect(result.length).toBeGreaterThan(0)
  })

  it('sanitizes reserved name with extension: NUL.flac → not NUL.flac', () => {
    expect(sanitizePathSegment('NUL.flac')).not.toBe('NUL.flac')
  })

  it('sanitizes reserved name with extension: CON.mp3 → not CON.mp3', () => {
    expect(sanitizePathSegment('CON.mp3')).not.toBe('CON.mp3')
  })

  it('preserves leading dots: ...And Justice For All starts with ...', () => {
    const result = sanitizePathSegment('...And Justice For All')
    expect(result).toMatch(/^\.\.\./)
  })

  it('strips trailing spaces: "track   " → "track"', () => {
    expect(sanitizePathSegment('track   ')).toBe('track')
  })

  it('returns _ for empty string', () => {
    expect(sanitizePathSegment('')).toBe('_')
  })

  it('returns _ for whitespace-only string', () => {
    expect(sanitizePathSegment('   ')).toBe('_')
  })
})

describe('atomicWriteJson + safeReadJson', () => {
  let tmpDir: string

  beforeEach(() => {
    tmpDir = mkdtempSync(join(tmpdir(), 'jms-test-'))
  })

  afterEach(() => {
    rmSync(tmpDir, { recursive: true, force: true })
  })

  it('round-trips JSON data correctly', () => {
    const filePath = join(tmpDir, 'data.json')
    const data = { artist: 'AC/DC', concurrentDownloads: 3, nested: { ok: true } }
    atomicWriteJson(filePath, data)
    expect(safeReadJson(filePath, null)).toEqual(data)
  })

  it('creates intermediate directories', () => {
    const filePath = join(tmpDir, 'a', 'b', 'c', 'data.json')
    atomicWriteJson(filePath, { ok: true })
    expect(safeReadJson(filePath, null)).toEqual({ ok: true })
  })

  it('returns fallback for a missing file', () => {
    const missing = join(tmpDir, 'does-not-exist.json')
    expect(safeReadJson(missing, 'fallback-value')).toBe('fallback-value')
  })

  it('returns fallback for corrupt JSON', () => {
    const filePath = join(tmpDir, 'corrupt.json')
    // Write valid JSON first, then corrupt it in-place (testing read path only)
    writeFileSync(filePath, '{this is not valid json', 'utf-8')
    expect(safeReadJson(filePath, null)).toBeNull()
  })

  it('does not leave .tmp file after successful write', () => {
    const filePath = join(tmpDir, 'data.json')
    atomicWriteJson(filePath, { x: 1 })
    const { existsSync } = require('fs')
    expect(existsSync(`${filePath}.tmp`)).toBe(false)
  })
})
```

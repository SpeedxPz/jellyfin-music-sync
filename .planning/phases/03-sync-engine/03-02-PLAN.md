// tests/lib/manifest.test.ts
import { describe, it, expect, beforeEach, afterEach } from 'vitest'
import { mkdtempSync, rmSync, writeFileSync, mkdirSync } from 'fs'
import { join } from 'path'
import { tmpdir } from 'os'
import {
  readManifest,
  writeManifest,
  needsDownload,
  isReferencedByOtherPlaylist,
  EMPTY_MANIFEST,
  type SyncManifest,
} from '../../src/main/lib/manifest'

describe('readManifest', () => {
  let tmpDir: string

  beforeEach(() => { tmpDir = mkdtempSync(join(tmpdir(), 'jms-manifest-')) })
  afterEach(() => { rmSync(tmpDir, { recursive: true, force: true }) })

  it('returns EMPTY_MANIFEST when no file exists', () => {
    const result = readManifest(tmpDir)
    expect(result).toEqual(EMPTY_MANIFEST)
  })

  it('returns EMPTY_MANIFEST for corrupt JSON (D-MANIFEST-ATOMIC)', () => {
    writeFileSync(join(tmpDir, '_jellyfin-sync.json'), '{bad json}', 'utf-8')
    const result = readManifest(tmpDir)
    expect(result).toEqual(EMPTY_MANIFEST)
  })

  it('returns EMPTY_MANIFEST for wrong version', () => {
    writeFileSync(
      join(tmpDir, '_jellyfin-sync.json'),
      JSON.stringify({ version: 2, items: {}, playlists: {} }),
      'utf-8'
    )
    const result = readManifest(tmpDir)
    expect(result).toEqual(EMPTY_MANIFEST)
  })

  it('round-trips a valid manifest', () => {
    const manifest: SyncManifest = {
      version: 1,
      items: {
        'item-1': { id: 'item-1', localPath: 'Artist/Album/track.flac', fileSize: 1234, syncedAt: '2026-01-01T00:00:00.000Z' },
      },
      playlists: {
        'pl-1': { id: 'pl-1', name: 'My Playlist', itemIds: ['item-1'] },
      },
    }
    writeManifest(tmpDir, manifest)
    const result = readManifest(tmpDir)
    expect(result).toEqual(manifest)
  })
})

describe('needsDownload', () => {
  let tmpDir: string

  beforeEach(() => { tmpDir = mkdtempSync(join(tmpdir(), 'jms-ndl-')) })
  afterEach(() => { rmSync(tmpDir, { recursive: true, force: true }) })

  it('returns true when item not in manifest (SYNC-05)', () => {
    expect(needsDownload(EMPTY_MANIFEST, 'item-X', tmpDir)).toBe(true)
  })

  it('returns true when file does not exist on disk', () => {
    const manifest: SyncManifest = {
      version: 1,
      items: { 'item-1': { id: 'item-1', localPath: 'A/B/t.flac', fileSize: 100, syncedAt: '' } },
      playlists: {},
    }
    expect(needsDownload(manifest, 'item-1', tmpDir)).toBe(true)
  })

  it('returns true when file size does not match manifest entry (SYNC-05)', () => {
    const fileDir = join(tmpDir, 'A', 'B')
    mkdirSync(fileDir, { recursive: true })
    const filePath = join(fileDir, 't.flac')
    writeFileSync(filePath, Buffer.alloc(200))  // 200 bytes on disk
    const manifest: SyncManifest = {
      version: 1,
      items: { 'item-1': { id: 'item-1', localPath: 'A/B/t.flac', fileSize: 100, syncedAt: '' } },
      playlists: {},
    }
    expect(needsDownload(manifest, 'item-1', tmpDir)).toBe(true)
  })

  it('returns false when file exists with matching size (SYNC-05)', () => {
    const fileDir = join(tmpDir, 'A', 'B')
    mkdirSync(fileDir, { recursive: true })
    const filePath = join(fileDir, 't.flac')
    writeFileSync(filePath, Buffer.alloc(100))  // 100 bytes
    const manifest: SyncManifest = {
      version: 1,
      items: { 'item-1': { id: 'item-1', localPath: 'A/B/t.flac', fileSize: 100, syncedAt: '' } },
      playlists: {},
    }
    expect(needsDownload(manifest, 'item-1', tmpDir)).toBe(false)
  })
})

describe('isReferencedByOtherPlaylist', () => {
  it('returns false when item appears only in current playlists (SYNC-06)', () => {
    const manifest: SyncManifest = {
      version: 1,
      items: {},
      playlists: {
        'pl-1': { id: 'pl-1', name: 'A', itemIds: ['item-1'] },
      },
    }
    expect(isReferencedByOtherPlaylist(manifest, 'item-1', ['pl-1'])).toBe(false)
  })

  it('returns true when item appears in a non-current playlist (D-DEL-SCOPE, SYNC-07)', () => {
    const manifest: SyncManifest = {
      version: 1,
      items: {},
      playlists: {
        'pl-1': { id: 'pl-1', name: 'A', itemIds: ['item-1'] },
        'pl-2': { id: 'pl-2', name: 'B', itemIds: ['item-1', 'item-2'] },
      },
    }
    // pl-1 is being synced; pl-2 is not in currentPlaylistIds
    expect(isReferencedByOtherPlaylist(manifest, 'item-1', ['pl-1'])).toBe(true)
  })

  it('returns false when item appears in no playlist (SYNC-06)', () => {
    expect(isReferencedByOtherPlaylist(EMPTY_MANIFEST, 'item-X', [])).toBe(false)
  })

  it('returns false when item is only in current playlist and there are other playlists (SYNC-07)', () => {
    const manifest: SyncManifest = {
      version: 1,
      items: {},
      playlists: {
        'pl-1': { id: 'pl-1', name: 'A', itemIds: ['item-1'] },
        'pl-2': { id: 'pl-2', name: 'B', itemIds: ['item-2'] },  // different item
      },
    }
    expect(isReferencedByOtherPlaylist(manifest, 'item-1', ['pl-1'])).toBe(false)
  })
})

// tests/lib/m3u8.test.ts
import { describe, it, expect } from 'vitest'
import { join } from 'path'
import { generateM3u8 } from '../../src/main/lib/m3u8'
import type { ManifestPlaylist } from '../../src/main/lib/manifest'
import type { M3u8ItemMeta } from '../../src/main/lib/m3u8'

describe('generateM3u8', () => {
  const destRoot = join('/', 'media', 'usb')  // posix-style for test portability

  it('starts with #EXTM3U header (M3U8-01)', () => {
    const playlist: ManifestPlaylist = { id: 'pl-1', name: 'Test', itemIds: [] }
    const result = generateM3u8(playlist, {}, destRoot)
    expect(result.startsWith('#EXTM3U')).toBe(true)
  })

  it('produces #EXTINF with duration in seconds from RunTimeTicks (M3U8-03)', () => {
    const playlist: ManifestPlaylist = { id: 'pl-1', name: 'Test', itemIds: ['item-1'] }
    const items: Record<string, M3u8ItemMeta> = {
      'item-1': {
        localPath: 'Artist/Album/track.flac',  // forward slashes
        runTimeTicks: 2_130_000_000,            // 213 seconds
        name: 'Track One',
      },
    }
    const result = generateM3u8(playlist, items, destRoot)
    expect(result).toContain('#EXTINF:213,Track One')
  })

  it('uses forward-slash relative paths (M3U8-02)', () => {
    const playlist: ManifestPlaylist = { id: 'pl-1', name: 'Test', itemIds: ['item-1'] }
    const items: Record<string, M3u8ItemMeta> = {
      'item-1': {
        localPath: 'Artist/Album/track.flac',
        runTimeTicks: 100_000_000,
        name: 'T',
      },
    }
    const result = generateM3u8(playlist, items, destRoot)
    // Path must use forward slashes, not backslashes
    expect(result).not.toContain('\\')
    expect(result).toContain('Artist/Album/track.flac')
  })

  it('omits tracks with no manifest entry (failed downloads)', () => {
    const playlist: ManifestPlaylist = { id: 'pl-1', name: 'Test', itemIds: ['item-1', 'item-missing'] }
    const items: Record<string, M3u8ItemMeta> = {
      'item-1': { localPath: 'A/B/t.flac', runTimeTicks: 100_000_000, name: 'T' },
      // item-missing intentionally absent
    }
    const result = generateM3u8(playlist, items, destRoot)
    const lines = result.split('\n').filter(Boolean)
    // Should have header + 2 lines for item-1 (EXTINF + path) = 3 total non-empty lines
    expect(lines.length).toBe(3)
  })

  it('rounds duration to nearest second (M3U8-03)', () => {
    const playlist: ManifestPlaylist = { id: 'pl-1', name: 'Test', itemIds: ['item-1'] }
    const items: Record<string, M3u8ItemMeta> = {
      'item-1': {
        localPath: 'A/B/t.flac',
        runTimeTicks: 3_550_000_000,  // 355 seconds exactly
        name: 'T',
      },
    }
    const result = generateM3u8(playlist, items, destRoot)
    expect(result).toContain('#EXTINF:355,T')
  })
})

// tests/lib/downloader.test.ts
import { describe, it, expect } from 'vitest'
import { join } from 'path'
import { buildDownloadUrl, buildLocalPath } from '../../src/main/lib/downloader'

describe('buildDownloadUrl', () => {
  const basePath = 'https://jellyfin.example.com'

  it('produces correct URL with static=true (SYNC-02)', () => {
    const url = buildDownloadUrl(basePath, 'abc123')
    expect(url).toBe('https://jellyfin.example.com/Audio/abc123/stream?static=true')
  })

  it('encodes item ID in URL', () => {
    const url = buildDownloadUrl(basePath, 'id with spaces')
    expect(url).toContain('/Audio/id%20with%20spaces/stream')
  })

  it('includes mediaSourceId when provided', () => {
    const url = buildDownloadUrl(basePath, 'abc123', 'src-456')
    expect(url).toContain('mediaSourceId=src-456')
  })

  it('does not include mediaSourceId when not provided', () => {
    const url = buildDownloadUrl(basePath, 'abc123')
    expect(url).not.toContain('mediaSourceId')
  })
})

describe('buildLocalPath', () => {
  const destRoot = join('/', 'media', 'usb')

  it('organizes files as dest/Artist/Album/Track (SYNC-03)', () => {
    const p = buildLocalPath(destRoot, 'Metallica', 'Black Album', 'Enter Sandman', 'flac')
    expect(p).toContain('Metallica')
    expect(p).toContain('Black Album')
    expect(p).toContain('Enter Sandman.flac')
  })

  it('sanitizes artist name with forward slash (SYNC-04 — AC/DC)', () => {
    const p = buildLocalPath(destRoot, 'AC/DC', 'Back in Black', 'Highway to Hell', 'mp3')
    // AC/DC → AC_DC after sanitization
    expect(p).not.toContain('AC/DC')
    expect(p).toContain('AC_DC')
  })

  it('sanitizes Windows reserved name in album (SYNC-04 — CON)', () => {
    const p = buildLocalPath(destRoot, 'Artist', 'CON', 'Track', 'flac')
    // CON is a Windows reserved name — sanitizePathSegment must rename it
    const parts = p.split(/[/\\]/)
    const albumPart = parts[parts.indexOf('Artist') + 1]
    expect(albumPart).not.toBe('CON')
  })

  it('handles ellipsis album name (SYNC-04 — ...And Justice For All)', () => {
    const p = buildLocalPath(destRoot, 'Metallica', '...And Justice For All', 'One', 'flac')
    expect(p).toContain('...And Justice For All')
  })

  it('uses Unknown Artist and Unknown Album for empty strings', () => {
    const p = buildLocalPath(destRoot, '', '', 'Track', 'flac')
    expect(p).toContain('Unknown Artist')
    expect(p).toContain('Unknown Album')
  })

  it('trims trailing spaces from track name before sanitization', () => {
    const p = buildLocalPath(destRoot, 'A', 'B', 'Track Name   ', 'flac')
    // Should not have trailing spaces or underscore replacement artifact before extension
    expect(p).toContain('Track Name.flac')
    expect(p).not.toContain('Track Name_   .flac')
  })
})
